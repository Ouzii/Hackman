17  /**
18   * Pelin grafiikoiden piirtäjä-luokka.
19   *
20   * @author Oce
21   */
22  public class Piirtaja extends JPanel implements Paivitettava {
23      
24      private Peli peli;
25      private int palikanKoko;
26      private Kayttoliittyma kali;
27  
28      /**
29       * Konstruktori piirtäjälle, joka asettaa tarvittavat yhteydet.
30       *
31       * @param peli Peli, jonka pohjalta grafiikat piirretään.
32       * @param palikanKoko Koko palikoille, jotka piirretään.
33       * @param kali Käyttöliittymä-luokka.
34       */
35      public Piirtaja(Peli peli, int palikanKoko, Kayttoliittyma kali) {
36          this.peli = peli;
37          this.palikanKoko = palikanKoko;
38          this.kali = kali;
39          this.setOpaque(true);
40          this.setBackground(Color.LIGHT_GRAY);
41      }
42      
43      private void piirraMenu(Graphics g) {
44          g.setColor(Color.BLACK);
45          g.drawString("Aloita painamalla <Enter>", 4 * this.palikanKoko, 7 * this.palikanKoko);
46          g.setColor(Color.BLUE);
47          g.drawString("Tulosta huipputulokset", 4 * this.palikanKoko, 9 * this.palikanKoko);
48          g.drawString("painamalla <F1>", 6 * this.palikanKoko, 10 * this.palikanKoko);
49          g.setColor(Color.BLACK);
50          g.drawString("Pelissä paina <R>", 4 * this.palikanKoko, 12 * this.palikanKoko);
51          g.drawString("aloittaaksesi alusta", 6 * this.palikanKoko, 13 * this.palikanKoko);
52          g.setColor(Color.BLUE);
53          g.drawString("Pelissä paina <P>", 4 * this.palikanKoko, 15 * this.palikanKoko);
54          g.drawString("pysäyttääksesi pelin", 6 * this.palikanKoko, 16 * this.palikanKoko);
55      }
56      
57      private void piirraHighscore(Graphics g) {
58          try {
59              this.peli.getHighscore().kirjoita();
60              g.setColor(Color.BLACK);
61              Scanner tiedostonLukija = new Scanner(new File("src/main/resources/highscore.txt"), "UTF-8");
62              int y = 4;
63              while (tiedostonLukija.hasNextLine()) {
64                  g.drawString(tiedostonLukija.nextLine(), 6 * this.palikanKoko, y * this.palikanKoko);
65                  y++;
66              }
67              g.setColor(Color.RED);
68              g.drawString("Paina <F1> palataksesi takaisin", 2 * this.palikanKoko, (y + 2) * this.palikanKoko);
69          } catch (Exception e) {
70              g.setColor(Color.RED);
71              g.drawString("Jokin meni pieleen, yritä uudelleen", this.palikanKoko, 10 * this.palikanKoko);
72              g.drawString("Paina <F1> palataksesi takaisin", 2 * this.palikanKoko, (12) * this.palikanKoko);
73          }
74      }
75      
76      private void piirraPelaaja(Graphics g) {
77          g.setColor(Color.GREEN);
78          if (this.peli.getPelaaja().isElossa()) {
79              g.fillOval(this.peli.getPelaaja().getX() * this.palikanKoko, this.peli.getPelaaja().getY() * this.palikanKoko, this.palikanKoko, this.palikanKoko);
80          }
81      }
82      
83      private void piirraBitit(Graphics g) {
84          g.setColor(Color.BLUE);
85          for (Bitti bitti : this.peli.getKartta().getBitit()) {
86              if (!bitti.isKeratty()) {
87                  g.fillOval(bitti.getX() * this.palikanKoko + 3, bitti.getY() * this.palikanKoko + 3, this.palikanKoko - 7, this.palikanKoko - 7);
88              }
89          }
90      }
91      
92      private void piirraViholliset(Graphics g) {
93          g.setColor(Color.RED);
94          g.fill3DRect(this.peli.getKartta().getVihuPun().getX() * this.palikanKoko, this.peli.getKartta().getVihuPun().getY() * this.palikanKoko, this.palikanKoko, this.palikanKoko, true);
95          
96          g.setColor(Color.BLACK);
97          g.fill3DRect(this.peli.getKartta().getVihuMus().getX() * this.palikanKoko, this.peli.getKartta().getVihuMus().getY() * this.palikanKoko, this.palikanKoko, this.palikanKoko, true);
98          
99          g.setColor(Color.YELLOW);
100         g.fill3DRect(this.peli.getKartta().getVihuKel().getX() * this.palikanKoko, this.peli.getKartta().getVihuKel().getY() * this.palikanKoko, this.palikanKoko, this.palikanKoko, true);
101         
102         g.setColor(Color.PINK);
103         g.fill3DRect(this.peli.getKartta().getVihuPin().getX() * this.palikanKoko, this.peli.getKartta().getVihuPin().getY() * this.palikanKoko, this.palikanKoko, this.palikanKoko, true);
104     }
105     
106     private void piirraSeinat(Graphics g) {
107         g.setColor(Color.GRAY);
108         
109         for (Palikka seina : this.peli.getKartta().getSeinat()) {
110             g.fill3DRect(seina.getX() * this.palikanKoko, seina.getY() * this.palikanKoko, this.palikanKoko, this.palikanKoko, true);
111         }
112     }
113 
114     /**
115      * Päämetodi piirtämiselle, joka suorittaa muut piirtometodit.
116      *
117      * @param g Grafiikanluonti Javassa.
118      */
119     @Override
120     public void paintComponent(Graphics g) {
121         super.paintComponent(g);
122         
123         Font suuri = new Font("Comic Sans MS", Font.BOLD, 34);
124         Font pieni = new Font("Comic Sans MS", Font.BOLD, 22);
125         g.setFont(pieni);
126         g.setColor(Color.LIGHT_GRAY);
127 //        g.fill3DRect(-20, -20, 40 * this.palikanKoko, 40 * this.palikanKoko, true);
128         if (!this.peli.getLogiikka().isAlkaa()) {
129             
130             if (!this.peli.getHighscore().isMenuun()) {
131                 this.piirraMenu(g);
132             } else {
133                 this.piirraHighscore(g);
134             }
135             
136         } else {
137             this.piirraSeinat(g);
138             this.piirraPelaaja(g);
139             this.piirraBitit(g);
140             this.piirraViholliset(g);
141             g.setColor(Color.RED);
142             g.drawString("PISTEET: " + this.peli.getLogiikka().getPojot(), 3 * this.palikanKoko - 20, 3 * this.palikanKoko - 41);
143             if (this.peli.getLogiikka().isVoita()) {
144                 g.setColor(Color.BLUE);
145                 g.setFont(suuri);
146                 g.drawString("VOITIT!", 6 * this.palikanKoko, 9 * this.palikanKoko);
147                 g.setFont(pieni);
148                 g.drawString("Paina <Space> siirtyäksesi", 3 * this.palikanKoko, 11 * this.palikanKoko);
149                 g.drawString("seuraavalle tasolle", 5 * this.palikanKoko, 12 * this.palikanKoko);
150             }
151             if (this.peli.getLogiikka().isHavia()) {
152                 g.setColor(Color.RED);
153                 g.setFont(suuri);
154                 g.drawString("HÄVISIT!", 6 * this.palikanKoko, 9 * this.palikanKoko);
155                 g.setFont(pieni);
156                 g.drawString("Paina <R> aloittaaksesi alusta", 2 * this.palikanKoko + 10, 11 * this.palikanKoko);
157             }
158         }
159     }

167     }
