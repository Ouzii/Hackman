23  public class Piirtaja extends JPanel implements Paivitettava {

25      private final Peli peli;
26      private final int palikanKoko;
27  
28      public Piirtaja(Peli peli, int palikanKoko) {
29          this.peli = peli;
30          this.palikanKoko = palikanKoko;
31      }
32  
33      private void piirraMenu(Graphics g) {
34          g.setColor(Color.BLACK);
35          g.drawString("Aloita painamalla <Enter>", 4 * this.palikanKoko, 7 * this.palikanKoko);
36          g.setColor(Color.BLUE);
37          g.drawString("Tulosta huipputulokset", 4 * this.palikanKoko, 9 * this.palikanKoko);
38          g.drawString("painamalla <F1>", 6 * this.palikanKoko, 10 * this.palikanKoko);
39          g.setColor(Color.BLACK);
40          g.drawString("Pelissä paina <R>", 4 * this.palikanKoko, 12 * this.palikanKoko);
41          g.drawString("aloittaaksesi alusta", 6 * this.palikanKoko, 13 * this.palikanKoko);
42          g.setColor(Color.BLUE);
43          g.drawString("Pelissä paina <P>", 4 * this.palikanKoko, 15 * this.palikanKoko);
44          g.drawString("pysäyttääksesi pelin", 6 * this.palikanKoko, 16 * this.palikanKoko);
45      }
46  
47      private void piirraHighscore(Graphics g) {
48          try {
49              g.setColor(Color.BLACK);
50              Scanner tiedostonLukija = new Scanner(new File("src/main/resources/highscore.txt"));
51              int y = 4;
52              while (tiedostonLukija.hasNextLine()) {
53                  g.drawString(tiedostonLukija.nextLine(), 8 * this.palikanKoko, y * this.palikanKoko);
54                  y++;
55              }
56              g.setColor(Color.RED);
57              g.drawString("Paina <F1> palataksesi takaisin", 2 * this.palikanKoko, (y + 2) * this.palikanKoko);
58          } catch (Exception e) {
59              g.setColor(Color.RED);
60              g.drawString("Jokin meni pieleen, yritä uudelleen", this.palikanKoko, 10 * this.palikanKoko);
61              g.drawString("Paina <F1> palataksesi takaisin", 2 * this.palikanKoko, (12) * this.palikanKoko);
62          }
63      }
64  
65      private void piirraPelaaja(Graphics g) {
66          g.setColor(Color.GREEN);
67          if (this.peli.getPelaaja().isElossa()) {
68              g.fill3DRect(this.peli.getPelaaja().getX() * this.palikanKoko, this.peli.getPelaaja().getY() * this.palikanKoko, this.palikanKoko, this.palikanKoko, true);
69          }
70      }
71  
72      private void piirraBitit(Graphics g) {
73          g.setColor(Color.BLUE);
74          for (Bitti bitti : this.peli.getKartta().getBitit()) {
75              if (!bitti.isKeratty()) {
76                  g.fillOval(bitti.getX() * this.palikanKoko + 3, bitti.getY() * this.palikanKoko + 3, this.palikanKoko - 7, this.palikanKoko - 7);
77              }
78          }
79      }
80  
81      private void piirraViholliset(Graphics g) {
82          g.setColor(Color.RED);
83          g.fill3DRect(this.peli.getKartta().getVihuPun().getX() * this.palikanKoko, this.peli.getKartta().getVihuPun().getY() * this.palikanKoko, this.palikanKoko, this.palikanKoko, true);
84  
85          g.setColor(Color.BLACK);
86          g.fill3DRect(this.peli.getKartta().getVihuMus().getX() * this.palikanKoko, this.peli.getKartta().getVihuMus().getY() * this.palikanKoko, this.palikanKoko, this.palikanKoko, true);
87  
88          g.setColor(Color.YELLOW);
89          g.fill3DRect(this.peli.getKartta().getVihuKel().getX() * this.palikanKoko, this.peli.getKartta().getVihuKel().getY() * this.palikanKoko, this.palikanKoko, this.palikanKoko, true);
90  
91          g.setColor(Color.PINK);
92          g.fill3DRect(this.peli.getKartta().getVihuPin().getX() * this.palikanKoko, this.peli.getKartta().getVihuPin().getY() * this.palikanKoko, this.palikanKoko, this.palikanKoko, true);
93      }
94  
95      private void piirraSeinat(Graphics g) {
96          g.setColor(Color.GRAY);
97  
98          for (Palikka seina : this.peli.getKartta().getSeinat()) {
99              g.fill3DRect(seina.getX() * this.palikanKoko, seina.getY() * this.palikanKoko, this.palikanKoko, this.palikanKoko, true);
100         }
101     }
102     /**
103      * Päämetodi piirtämiselle, joka suorittaa muut piirtometodit.
104      * @param g 
105      */
106     @Override
107     public void paintComponent(Graphics g) {
108         super.paintComponent(g);
109         Font suuri = new Font("Comic Sans MS", Font.BOLD, 34);
110         Font pieni = new Font("Comic Sans MS", Font.BOLD, 22);
111         g.setFont(pieni);
112         g.setColor(Color.LIGHT_GRAY);
113         g.fill3DRect(-20, -20, 40 * this.palikanKoko, 40 * this.palikanKoko, true);
114         if (!this.peli.getLogiikka().isAlkaa()) {
115 
116             if (!this.peli.getLogiikka().isHighscore()) {
117                 this.piirraMenu(g);
118             } else {
119                 this.piirraHighscore(g);
120             }
121 
122         } else {
123 
124             this.piirraPelaaja(g);
125             this.piirraBitit(g);
126             this.piirraViholliset(g);
127             this.piirraSeinat(g);
128             g.setColor(Color.RED);
129             g.drawString("PISTEET: " + this.peli.getLogiikka().getPojot(), 3 * this.palikanKoko - 20, 3 * this.palikanKoko - 41);
130             if (this.peli.getLogiikka().isVoita()) {
131                 g.setColor(Color.BLUE);
132                 g.setFont(suuri);
133                 g.drawString("VOITIT!", 6 * this.palikanKoko, 9 * this.palikanKoko);
134                 g.setFont(pieni);
135                 g.drawString("Paina <Space> siirtyäksesi", 3 * this.palikanKoko, 11 * this.palikanKoko);
136                 g.drawString("seuraavalle tasolle", 5 * this.palikanKoko, 12 * this.palikanKoko);
137             }
138             if (this.peli.getLogiikka().isHavia()) {
139                 g.setColor(Color.RED);
140                 g.setFont(suuri);
141                 g.drawString("HÄVISIT!", 6 * this.palikanKoko, 9 * this.palikanKoko);
142             }
143         }
144     }

147      * @return 
148      */
149     @Override
150     public boolean paivita() {
151         repaint();
152         return true;
153     }
