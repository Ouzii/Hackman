8   /**
9    * Käyttöliittymä, joka huolehtii pelin valikoista ja käynnistyksestä.
10   *
11   * @author oce
12   */
13  public class Kayttoliittyma implements Runnable {
14  
15      private JFrame frame;
16      private Peli peli;
17      private final int sivunPituus;
18      private Piirtaja piirto;
19      public int pojot;
20      private String nimi;
21      private JButton nappi;
22      private JTextField tekstikentta;
23      private JLabel merkinta;
24  
25      /**
26       * Konstruktori käyttöliittymälle, joka luo uuden pelin ja asettaa oikean
27       * mittasuhteen pelille.
28       *
29       * @param sivunPituus Annettava koko palikoille pelissä.
30       * @param error Kertoo käynnistetäänkö error-moodissa.
31       */
32      public Kayttoliittyma(int sivunPituus, boolean error) {
33          this.sivunPituus = sivunPituus;
34          this.peli = new Peli(20, 20, new Kartta1(20, 20));
35          this.pojot = 0;
36          this.nimi = "";
37          this.nappi = new JButton("Kirjaudu");
38          this.tekstikentta = new JTextField("");
39          if (!error) {
40              this.merkinta = new JLabel("Anna nimesi: (max. 20 merkkiä)", SwingConstants.CENTER);
41          } else {
42              this.merkinta = new JLabel("Liian pitkä nimi!", SwingConstants.CENTER);
43          }
44          this.asetaUlkoasu();
45      }
46  
47      private void asetaUlkoasu() {
48          Font fontti = new Font("Comic Sans MS", Font.BOLD, 34);
49          Font fontti2 = new Font("Comic Sans MS", Font.BOLD, 24);
50          Dimension dimenssio = new Dimension(150, 150);
51          this.merkinta.setFont(fontti2);
52          this.nappi.setFont(fontti);
53          this.tekstikentta.setFont(fontti);
54          this.merkinta.setOpaque(true);
55          this.merkinta.setBackground(Color.BLACK);
56          this.merkinta.setForeground(Color.WHITE);
57          this.merkinta.setPreferredSize(dimenssio);
58          this.tekstikentta.setOpaque(true);
59          this.tekstikentta.setBorder(null);
60          this.tekstikentta.setBackground(Color.BLACK);
61          this.tekstikentta.setForeground(Color.RED);
62          this.tekstikentta.setPreferredSize(dimenssio);
63          this.tekstikentta.setHorizontalAlignment(SwingConstants.CENTER);
64          this.nappi.setOpaque(true);
65          this.nappi.setBackground(Color.WHITE);
66          this.nappi.setForeground(Color.BLACK);
67          this.nappi.setPreferredSize(dimenssio);
68      }
69  
70      public String getNimi() {
71          return nimi;
72      }

74      /**
75       * Asettaa JTextFieldistä saadun merkkijonon pelaajan nimeksi
76       * Highscore-luokkaan.
77       *
78       * @param nimi pelaajan antama nimi.
79       */
80      public void setNimi(String nimi) {
81          this.nimi = nimi;
82          this.peli.getHighscore().setNimi(nimi);
83      }
84  
85      /**
86       * Käynnistää avattavan ikkunan.
87       */
88      @Override
89      public void run() {
90          frame = new JFrame("HACKMAN");
91          int leveys = (peli.getLeveys()) * (sivunPituus + 2) - 4;
92          int korkeus = (peli.getKorkeus()) * (sivunPituus + 3) - 2;
93          frame.setPreferredSize(new Dimension(leveys, korkeus));
94          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
95          BorderLayout l = new BorderLayout();
96          frame.setLayout(l);
97          this.nappi.addActionListener(new NapinKuuntelija(this.peli, this.tekstikentta, this));
98          frame.add(this.merkinta, BorderLayout.NORTH);
99          frame.add(this.tekstikentta, BorderLayout.CENTER);
100         frame.add(this.nappi, BorderLayout.SOUTH);
101         frame.pack();
102         frame.setVisible(true);
103     }
104 
105     public JFrame getFrame() {
106         return frame;
107     }
108 
109     /**
110      * Luo piirtäjän ikkunalle ja lisää ikkunalle näppäimistönkuuntelijan.
111      *
112      * @param container Piirtoalue.
113      */
114     public void luoKomponentit(Container container) {
115         frame.remove(this.merkinta);
116         frame.remove(this.nappi);
117         frame.remove(this.tekstikentta);
118         this.piirto = new Piirtaja(this.peli, this.sivunPituus, this);
119         this.peli.setPaivitettava(piirto);
120         container.add(piirto);
121         NappaimistonKuuntelija nk = new NappaimistonKuuntelija(this.peli.getPelaaja(), this.peli, this);
122         frame.addKeyListener(nk);
123     }
124 
125     public Paivitettava getPaivitettava() {
126         return this.piirto;
127     }

137     /**
138      * Luo uuden pelin seuraavalla kartalla, jos edellinen kartta voitettu.
139      */
140     public void nextMap() {
141         frame.remove(piirto);
142         frame.removeKeyListener(frame.getKeyListeners()[0]);
143         this.pojot = this.peli.getLogiikka().getPojot();
144         if (this.peli.getKartta().toString().equals("Kartta1")) {
145             this.peli = new Peli(20, 20, new Kartta2(20, 20));
146             this.peli.getLogiikka().setPojot(this.pojot);
147             this.peli.getLogiikka().setAlkaa();
148         } else if (this.peli.getKartta().toString().equals("Kartta2")) {
149             this.peli = new Peli(20, 20, new Kartta3(20, 20));
150             this.peli.getLogiikka().setPojot(this.pojot);
151             this.peli.getLogiikka().setAlkaa();
152         } else if (this.peli.getKartta().toString().equals("Kartta3")) {
153             this.peli = new Peli(20, 20, new Kartta4(20, 20));
154             this.peli.getLogiikka().setPojot(this.pojot);
155             this.peli.getLogiikka().setAlkaa();
156         } else if (this.peli.getKartta().toString().equals("Kartta4")) {
157             this.peli = new Peli(20, 20, new Kartta5(20, 20));
158             this.peli.getLogiikka().setPojot(this.pojot);
159             this.peli.getLogiikka().setAlkaa();
160         } else if (this.peli.getKartta().toString().equals("Kartta5")) {
161             this.uusiPeli();
162             this.menu();
163             this.peli.getHighscore().setMenuun(true);
164             this.peli.getHighscore().onkoHighscore(pojot);
165             this.peli.getHighscore().kirjoita();
166             return;
167         }
168         this.luoKomponentit(frame);
169         frame.pack();
170         frame.setVisible(true);
171     }
172 
173     /**
174      * Luo uuden pelin ensimmäisellä kartalla.
175      *
176      * @return Peli palauttaa uuden pelin.
177      */
178     public Peli uusiPeli() {
179         if (this.piirto != null) {
180             frame.remove(piirto);
181         }
182         if (frame.getKeyListeners().length != 0) {
183             frame.removeKeyListener(frame.getKeyListeners()[0]);
184         }
185         this.peli = new Peli(20, 20, new Kartta1(20, 20), this.nimi);
186         this.peli.getLogiikka().setAlkaa();
187         this.luoKomponentit(frame);
188         frame.pack();
189         frame.setVisible(true);
190         return this.peli;
191     }
